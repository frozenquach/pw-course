Git
-git commit --amend -m "new commit" //Thay đổi message commit mới nhất
-git restore --staged <file> // đưa 1 file từ staging về working area
-git reset --soft HEAD~1 //Đưa nội dung commit cuối (HEAD~1) về staging 
-git reset HEAD~1 //Đưa nội dung commit cuối (HEAD~1) về working area
Git branch
-git branch <branch name> //Tạo 1 branch có tên  <branch name>
-git checkout <branch name> //Chuyển qua branch <branch name>
-git checkout -b <branch name> //Tạo và chuyển qua <branch name>
-git branch -D <branch name> //Xóa <branch name> (note: nên qua nhánh khác để xóa)
-git push origin <branch name> //Push nhánh <branch name> lên git remote
-git push -D <branch name> //Xóa <branch name> trên git remote
